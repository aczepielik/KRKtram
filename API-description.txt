### TTSS API ###

1. Położenie tramawajów

http://www.ttss.krakow.pl/internetservice/geoserviceDispatcher/services/vehicleinfo/vehicles

Struktura:

lastUpdate: czas
vehicles:
  id,
  
  #opcja 1
  isDeleted: true #w większości id z początku listy
  
  #opcja 2
  category: tram #innych chyba nie ma
  color: 0x00000 #inncyh chyba nie ma
  tripId
  name
  #opcja 2.1
  longitude
  latitude
  heding
  #opcja 2.2
  path
    length
    y1
    y2
    x2
    angle
    x1
     longitude
  latitudeas
  heding
